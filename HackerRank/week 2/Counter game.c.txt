#include <stdio.h>
const char* counterGame(unsigned long long n) {
int turns = 0;
while (n != 1) {
    if ((n & (n-1)) == 0) {
        n >>= 1;
    } else {
        unsigned long long p = 1ULL << (63 - __builtin_clzll(n));
        n -= p;
    }
    turns++;
    }    
    return (turns % 2 == 1) ? "Louise" : "Richard";
}
int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        unsigned long long n;
        scanf("%llu", &n);
        printf("%s\n", counterGame(n));
    }
    return 0;
}
