#include <stdio.h>
#include <string.h>
#include <ctype.h>
void caesarCipher(char s[], int k) {
    int n = strlen(s);
    k = k % 26;
    for (int i = 0; i < n; i++) {
        if (isalpha(s[i])) {
            char base = isupper(s[i]) ? 'A' : 'a';
            s[i] = ((s[i] - base + k) % 26) + base;
        }
    }
}
int main() {
    int n, k;
    char s[105];
    scanf("%d", &n);
    scanf("%s", s);
    scanf("%d", &k);
    caesarCipher(s, k);
    printf("%s\n", s);
    return 0;
}
