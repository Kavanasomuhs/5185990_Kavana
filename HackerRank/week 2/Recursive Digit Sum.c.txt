#include <stdio.h>
#include <string.h>
long long superDigitCalc(long long num) {
    if (num < 10) return num;
    long long sum = 0;
    while (num > 0) {
        sum += num % 10;
        num /= 10;
    }
    return superDigitCalc(sum);
}
long long superDigit(char n[], long long k) {
    long long sum = 0;
    for (int i = 0; n[i] != '\0'; i++) {
        sum += (n[i] - '0');
    }
    sum *= k;
    return superDigitCalc(sum);
}
int main() {
    char n[100005];
    long long k;
    scanf("%s %lld", n, &k);
    printf("%lld\n", superDigit(n, k));
    return 0;
}
