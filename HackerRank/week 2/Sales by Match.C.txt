#include <stdio.h>
int sockMerchant(int n, int ar[]) {
    int count[101] = {0};
    int pairs = 0;
    
    for (int i = 0; i < n; i++) {
        count[ar[i]]++;
    }
    for (int i = 0; i <= 100; i++) {
        pairs += count[i] / 2;
    }
    return pairs;
}
int main() {
    int n;
    scanf("%d", &n);
    int ar[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &ar[i]);
    }
    printf("%d\n", sockMerchant(n, ar));
    return 0;
}
