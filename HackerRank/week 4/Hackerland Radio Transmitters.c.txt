#include <stdio.h>
#include <stdlib.h>

int cmpfunc(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int hackerlandRadioTransmitters(int x_count, int* x, int k) {
        qsort(x, x_count, sizeof(int), cmpfunc);

    int transmitters = 0;
    int i = 0;

    while (i < x_count) {
        transmitters++;

       
        int loc = x[i] + k;
        while (i < x_count && x[i] <= loc) {
            i++;
        }

                int pos = x[i - 1];
        loc = pos + k;

        
        while (i < x_count && x[i] <= loc) {
            i++;
        }
    }

    return transmitters;
}

int main() {
    int n, k;
    scanf("%d %d", &n, &k);

    int *x = (int*)malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) {
        scanf("%d", &x[i]);
    }

    int result = hackerlandRadioTransmitters(n, x, k);
    printf("%d\n", result);

    free(x);
    return 0;
}
