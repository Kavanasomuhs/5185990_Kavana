#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int cmp(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

char* isValid(char* s) {
    static char result[4]; 
    int freq[26] = {0};
    int arr[26], count = 0;

   
    for (int i = 0; s[i]; i++) {
        freq[s[i] - 'a']++;
    }

    
    for (int i = 0; i < 26; i++) {
        if (freq[i] > 0) arr[count++] = freq[i];
    }

    
    qsort(arr, count, sizeof(int), cmp);

        if (arr[0] == arr[count - 1]) {
        strcpy(result, "YES");
        return result;
    }

        if (arr[0] == 1 && arr[1] == arr[count - 1]) {
        strcpy(result, "YES");
        return result;
    }

    
    if (arr[0] == arr[count - 2] && arr[count - 1] == arr[count - 2] + 1) {
        strcpy(result, "YES");
        return result;
    }

    strcpy(result, "NO");
    return result;
}

int main() {
    char s[100005];
    scanf("%s", s);
    printf("%s\n", isValid(s));
    return 0;
}
