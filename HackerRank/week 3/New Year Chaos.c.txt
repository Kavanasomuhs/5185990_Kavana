#include <stdio.h>

void minimumBribes(int q[], int n) {
    int bribes = 0;
    for (int i = 0; i < n; i++) {
        // Check if any person has moved more than 2 positions ahead
        if (q[i] - (i + 1) > 2) {
            printf("Too chaotic\n");
            return;
        }
    }
    // Count total bribes
    for (int i = 0; i < n; i++) {
        // A person can only be bribed by someone ahead of them who started
        // at most two positions before their current position
        int start = q[i] - 2;
        if (start < 0) start = 0;
        for (int j = start; j < i; j++) {
            if (q[j] > q[i]) bribes++;
        }
    }
    printf("%d\n", bribes);
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        int q[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &q[i]);
        }
        minimumBribes(q, n);
    }
    return 0;
}
