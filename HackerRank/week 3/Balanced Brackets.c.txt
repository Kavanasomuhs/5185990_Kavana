char* isBalanced(char* s) {
#include <string.h>
    static char yes[] = "YES";
    static char no[]  = "NO";

    char stack[1000];
    int top = -1;

    for (int i = 0; s[i] != '\0'; i++) {
        char ch = s[i];

                if (ch == '(' || ch == '{' || ch == '[') {
            stack[++top] = ch;
        }
                else if (ch == ')' || ch == '}' || ch == ']') {
            if (top == -1) return no; 
            char topChar = stack[top--];

            if ((ch == ')' && topChar != '(') ||
                (ch == '}' && topChar != '{') ||
                (ch == ']' && topChar != '[')) {
                return no; // Mismatch
            }
        }
    }

    return (top == -1) ? yes : no;
}
